                                            DATABASE DOCUMENTATION
                                            ======================

1) user
    ===
        1.1) _id :
                type : objectid
                description : unique id to identify user
        1.2) firstname :
                type : string
                description : firstname of the user
        1.3) lastname :
                type : string
                description : lastname of the user
        1.4) phonenumber :
                type : string
                description : phonenumber of the user
        1.5) email :
                type : string
                description : email address of the user
        1.6) password :
                type : string
                description : hashed password to authenticate user
        1.7) cartid :
                type : objectid
                refer : cart collection
                description : id to uniquely identify user cart from cart collection
        1.8) purchaseids :
                type : array
                description : array contain all previous purchase ids
                parameters :
                    type : objectid
                    refer : purchasehistory collection
                    description : id to uniquely identify purchase history
        1.9) address :
                type : array
                description : array contain all saved addressess
                parameters :
                    _id :
                        type : objectid
                        description : id to uniquely identify address from address array
                    feature : 
                        type : string
                        description : house name or house number
                    landmark :
                        type : string
                        description : landmark of the address
                    addressline : 
                        type : string
                        description : address line
                    coordinates :
                        type : array
                        description : lattitude and longitude of the address
                        parameters :
                            lng :
                                type : double
                                description : longitude of the address
                            lat :
                                type : double
                                description : lattitude of the address
                    isfavourite :
                        type : boolean
                        description : whether to add this address to favourite list or not
                    isdefault :
                        type : boolean
                        description : to make this address default
        1.10) shopinfoids : 
                type : array
                description : array contain all saved shop ids
                parameters :
                    type : objectid
                    description : id of shop that these user works
        1.11) notification :
                type : array
                description : notifications
                parameters :
                    _id : 
                        type : objectid
                        description : an id to uniquely identify notification
                    type :
                        type : string
                        description : type of the notification
                    shopinfoid
                        type : objectid
                        description : shop id (this is for employeee verification)

        -----------------------------------------------------------------------------------------------------------------------------
        *****************************************************************************************************************************

2) session 
    ======
        2.1) _id :
                type : objectid
                description : id to uniquely identify session of the user
        2.2) userid :
                type : objectid
                refer : user collection
                description : id of the user of this session
        2.3) transaction :
                type : object
                description : details of the transaction(payment)
                parameters :
                    temporaryproducts :
                        type : array
                        description : array of the products that are going to purchase
                        parameters :
                            _id :
                                type : objectid
                                description : id to uniquely identify product from temporaryproducts array
                            uniqueid :
                                type : string
                                description : uniqueid of the product like barcode
                            shopinfoid :
                                type : objectid
                                refer : shopinfo collection
                                description : id of the shop
                            productsid :
                                type : objectid
                                refer : products collection
                                description : id of the product
                            quantity :
                                type : number
                                description : quantity of the product
                    purchaseid :
                        type : objectid
                        refer : purchasehistory collection
                        description : purchase id to uniquely identify purchase summary from purchasehistory
                    paymentintentid :
                        type : string
                        description : stripe paymentintent id to uniquely identify payment status
